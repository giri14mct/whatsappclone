import React,{useState} from 'react'
import './Chat.css';

import MoreVertIcon from '@material-ui/icons/MoreVert';
import SearchIcon from '@material-ui/icons/Search';
import AttachmentIcon from '@material-ui/icons/Attachment';
import SendIcon from '@material-ui/icons/Send';


const Chat = () => {
    const [chat,setChat] = useState('');
    const [display, setDisplay] = useState([]);
  
    let store = (e) => {
        setChat(e.target.value);
    }

    let displayer = (e)  =>{
        e.preventDefault();
        if(chat){
            setDisplay((data)=> [...data,chat]);
        }
        setChat("");
        console.log(chat);  
    }
    let openFile = () => {
        document.getElementById("file").click();
        
    }
   
    return (
        <>
        <div className="chat-head">
           <header className="header-content">
                 <img className="pic" src ="https://www.w3schools.com/w3images/avatar2.png" alt="Profile of your fried" />
               <div className="head-profile" >
                   <div className="profile-name" >jawa</div>
                   <div className="profile-time">last seen today at 6:16pm</div>
               </div>
               <div className="chat-icon" >
                   <MoreVertIcon />
                   <SearchIcon />
               </div>
           </header>
           <main className="message-area">
                <div className ="message " >  
               
                         {display.map((msg) =>(
                             
                            <div className="messageItem  messageItem-none">
                                <li className="message-text">{msg}</li>
                                <p className="message-time">{  new Date().toLocaleTimeString()}</p>
                             
                             </div> 
                         ))}    
                </div>
         
           </main>
           <form onSubmit={displayer}  className="input-area">
                   <input type="file"  id="file" multiple style={{display:"none"}} />
                   <AttachmentIcon onClick={openFile} className ="input-Attachment" style={{color:'#919191'}} />
                       
                   <input name="chat"  className="input-text" value={chat} onChange={store} type="text"   placeholder="Type a message" />

                   <button type="submit" value={display} className="input-send" ><SendIcon style={{color:'#919191'}} /></button>
              
           </form>
        </div> 
        </>
    )
}

export default Chat
